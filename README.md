# sky-immo
# Install JSON Server

$npm install -g json-server

# Clone repository
$git clone https://github.com/baytulizza/sky-immo.git

# Start JSON Server
.
├── json-server
│   └── db.json
└── README.md

$cd ./sky-immo/json-server
$json-server --watch db.json
